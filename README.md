# Инструкция:
1. sudo apt update
2. sudo apt upgrade
3. sudo apt install gcc make
4. sudo apt autoremove
5. wget https://github.com/baklanomax/easypoetry/archive/refs/tags/v0.1.tar.gz
6. tar -zxf v0.1.tar.gz
7. rm v0.1.tar.gz
8. cd easypoetry-0.1/
9. make
10. sudo make install
